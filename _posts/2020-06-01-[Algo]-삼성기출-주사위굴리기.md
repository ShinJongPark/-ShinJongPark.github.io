~~~java
package sw;

import java.io.*;

public class Main {
	
	static int N,M,r,c,K;
	static int[][] ball, map;
	static int[] controller;
	static int[][] dir = {{},{0,1},{0,-1},{-1,0},{1,0}}; 
	static boolean check(int x, int y) {
		if(x>=0 && x<map.length && y>=0 && y<map[x].length)
			return true;
		else return false;
	}
	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		String[] input = br.readLine().trim().split(" ");
		
		N = Integer.parseInt(input[0]);
		M = Integer.parseInt(input[1]);
		r = Integer.parseInt(input[2]);
		c = Integer.parseInt(input[3]);
		K = Integer.parseInt(input[4]);
		
		map = new int[N][M];
		ball = new int[4][3];
		controller = new int[K];
		/*
		 * 
		 */
		ball[3][1] = map[r][c];
		map[r][c] = 0;
		for(int i=0; i<N; i++) {
			input = br.readLine().trim().split(" ");
			for(int j=0; j<M; j++) {
				map[i][j] = Integer.parseInt(input[j]);
			}
		}
		input = br.readLine().trim().split(" ");
		int dx, dy, k;
		for(int i=0; i<K; i++) {
			k = Integer.parseInt(input[i]);
			dx = r + dir[k][0];
			dy = c + dir[k][1];
			if(check(dx,dy)) {
				if(k == 1) {
					goRight();
				}else if(k == 2) {
					goLeft();
				}else if(k == 3) {
					goUp();
				}else if(k == 4) {
					goDown();
				}
				if(map[dx][dy] == 0) {
					map[dx][dy] = ball[3][1];
				}else {
					ball[3][1] = map[dx][dy];
					map[dx][dy] = 0;
				}
				r = dx;
				c = dy;
				System.out.println(ball[1][1]);
			}
		}
	}
	
	static void goRight() {
		int tmp = ball[3][1];
		ball[3][1] = ball[1][2];
		ball[1][2] = ball[1][1];
		ball[1][1] = ball[1][0];
		ball[1][0] = tmp;
	}
	
	static void goDown() {
		int tmp = ball[3][1];
		ball[3][1] = ball[2][1];
		ball[2][1] = ball[1][1];
		ball[1][1] = ball[0][1];
		ball[0][1] = tmp;
	}

	static void goLeft() {
		int tmp = ball[3][1];
		ball[3][1] = ball[1][0];
		ball[1][0] = ball[1][1];
		ball[1][1] = ball[1][2];
		ball[1][2] = tmp;
	}
	
	static void goUp() {
		int tmp = ball[3][1];
		ball[3][1] = ball[0][1];
		ball[0][1] = ball[1][1];
		ball[1][1] = ball[2][1];
		ball[2][1] = tmp;
	}
	
	static void print() {
		for(int i=0; i<N; i++) {
			for(int j=0; j<M; j++) {
				System.out.print(map[i][j] + " ");
			}
			System.out.println();
		}
	}
}
~~~
